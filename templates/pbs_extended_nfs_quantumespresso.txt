[cluster QuantumESPRESSO]
FormLayout = selectionpanel
Category = Applications
IconUrl = https://raw.githubusercontent.com/hirtanak/scripts/master/QuantumESPRESSlogo.png

Autoscale = $Autoscale

    [[node defaults]]
    UsePublicNetwork = $UsePublicNetwork
    Credentials = $Credentials    
    SubnetId = $SubnetId
    Region = $Region
    KeyPairLocation = ~/.ssh/cyclecloud.pem
#    KeyPairLocation = ~/examples/id_rsaccuser 
   
        [[[cluster-init QuantumESPRESSO:default:1.0.0]]]
        Optional = True

        [[[configuration]]]
	QE_VERSION = $configuration_QE_VERSION
        QE_DL_URL = $configuration_QE_DL_URL
	LICENSE = $configuration_LICENSE
        KEY = $configuration_KEY

    [[node master]]
    MachineType = $MasterMachineType
    IsReturnProxy = $ReturnProxy
    AdditionalClusterInitSpecs = $MasterClusterInitSpecs
    ImageName = $ImageName

        [[[configuration]]]

        [[[cluster-init QuantumESPRESSO:master:1.0.0]]]
        Optional = True

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $UsePublicNetwork
	StaticPublicIpAddress = true

        [[[input-endpoint ganglia]]]
        PrivatePort = 8652
        PublicPort = 8652

        # Add 2 premium disks in a RAID 0 configuration to the NFS export
        [[[volume nfs-1]]]
        Size = 512
        SSD = True
        Mount = nfs
        Persistent = true

        [[[volume nfs-2]]]
        Size = 512
        SSD = True
        Mount = nfs
        Persistent = true

	[[[configuration cyclecloud.mounts.nfs]]]
	mountpoint = /mnt/exports/apps
	fs_type = ext4
	raid_level = 0

        [[[configuration cyclecloud.exports.nfs_data]]]
        type = nfs
        export_path = /mnt/exports/apps

    [[nodearray execute]]
    MachineType = $ExecuteMachineType
    MaxCoreCount = $MaxExecuteCoreCount
    Interruptible = $UseLowPrio
    AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs
    InstallJetpack = true
    JetpackPlatform = centos-7
    ImageName = $ImageNameExecute
    Azure.SingleScaleset = true

        [[[configuration]]]

        [[[cluster-init QuantumESPRESSO:execute:1.0.0]]]
        Optional = True

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic

        [[[configuration cyclecloud.mounts.nfs_data]]]
        type = nfs
        mountpoint = /mnt/exports/apps
        export_path = /mnt/exports/apps

    [[nodearray viz-vmss]]
    MachineType = $VizMachineType
    Interruptible = $UseLowPrio
    InstallJetpack = true
    JetpackPlatform = windows
    AwaitInstallation = flase
    ImageName = $ImageNameViz

    Username = $Username
    Password = $WindowsPassword

        [[[configuration]]]

        [[[cluster-init QuantumESPRESSO:viz:1.0.0]]]
        Optional = True

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $VizNodesPublic
        StaticPublicIpAddress = true

    [[node viz-vm]]
    MachineType = $VizMachineType
    Interruptible = $UseLowPrio
    InstallJetpack = true
    JetpackPlatform = windows
    AwaitInstallation = flase
    ImageName = $ImageNameViz

    Username = $Username
    Password = $WindowsPassword

        [[[configuration]]]

        [[[cluster-init QuantumESPRESSO:viz:1.0.0]]]
        Optional = True

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $VizNodesPublic
        StaticPublicIpAddress = true
